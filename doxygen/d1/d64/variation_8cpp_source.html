45class InsertMutation : public MutationBase
46{
47public:
48 template<Brush::ProgramType T>
49 static auto find_spots(Program<T>& program, Variation<T>& variator,
50 const Parameters& params)
51 {
52 vector<float> weights;
53
54 if (program.Tree.size() < params.get_max_size()) {
55 Iter iter = program.Tree.begin();
56 std::transform(program.Tree.begin(), program.Tree.end(), std::back_inserter(weights),
57 [&](const auto& n){
58 size_t d = 1+program.Tree.depth(iter);
59 std::advance(iter, 1);
60
61 // check if SS holds an operator to avoid failing `check` in sample_op_with_arg
62 if ((d >= params.get_max_depth())
63 || (variator.search_space.node_map.find(n.ret_type) == variator.search_space.node_map.end())) {
64 return 0.0f;
65 }
66 else {
67 return n.get_prob_change();
68 }
69 });
70 }
71 else {
72 // fill the vector with zeros, since we're already at max_size
73 weights.resize(program.Tree.size());
74 std::fill(weights.begin(), weights.end(), 0.0f);
75 }
76

80 template<Brush::ProgramType T>
81 static auto mutate(Program<T>& program, Iter spot, Variation<T>& variator,
82 const Parameters& params)
83 {
84 auto spot_type = spot.node->data.ret_type;
85
86 // pick a random compatible node to insert (with probabilities given by
87 // node_map_weights). The `-1` represents the node being inserted.
88 // Ideally, it should always find at least one match (the same node
89 // used as a reference when calling the function). However, we have a
90 // size restriction, which will be relaxed here (just as it is in the PTC2
91 // algorithm). This mutation can create a new expression that exceeds the
92 // maximum size by the highest arity among the operators.
93
94 std::optional<Node> n = variator.bandit_sample_op_with_arg(
95 spot_type, spot_type, params.max_size-program.Tree.size()-1);
96
97 if (!n) // there is no operator with compatible arguments
98 return false;
99
100 // make node n wrap the subtree at the chosen spot
101 auto parent_node = program.Tree.wrap(spot, *n);
102
103 // now fill the arguments of n appropriately
104 bool spot_filled = false;
105 for (auto a: (*n).arg_types)
106 {
107 if (spot_filled)
108 {
109 // if spot is in its child position, append children.
110 auto opt = variator.bandit_sample_terminal(a);
111
112 if (!opt)
113 return false;
114
115 program.Tree.append_child(parent_node, opt.value());
116 }
117 // if types match, treat this spot as filled by the spot node
118 else if (a == spot_type)
119 spot_filled = true;
120 // otherwise, add siblings before spot node
121 else {
122 auto opt = variator.bandit_sample_terminal(a);
123
124 if (!opt)
125 return false;
126
127 program.Tree.insert(spot, opt.value());
128 }
129 }
130
131 return true;
132 }

300 template<Brush::ProgramType T>
301 static auto mutate(Program<T>& program, Iter spot, Variation<T>& variator,
302 const Parameters& params)
303 {
304 // check if we exceeded the size/depth constrains (without subtracting,
305 // to avoid overflow cases if the user sets max_size smaller than arity
306 // of smallest operator. The overflow would happen when calculating d and
307 // s in the following lines, to choose the PTC2 limits)
308 if ( params.max_size <= (program.Tree.size() - program.Tree.size(spot))
309 || params.max_depth <= program.Tree.depth(spot) )
310 return false;
311
312 auto spot_type = spot.node->data.ret_type;
313
314 // d and s must be compatible with PTC2 --- they should be based on
315 // tree structure, not program structure
316 size_t d = params.max_depth - program.Tree.depth(spot);
317 size_t s = params.max_size - (program.Tree.size() - program.Tree.size(spot));
318
319 s = r.rnd_int(1, s+1);
320
321 // sample subtree uses PTC2, which operates on depth and size of the tree<Node>
322 // (and not on the program!). we shoudn't care for weights here
323
324 auto subtree = variator.search_space.sample_subtree(spot.node->data, d, s);
325
326
327 if (!subtree) // there is no terminal with compatible arguments
328 return false;
329
330
331 // if optional contains a Node, we access its contained value
332 program.Tree.erase_children(spot);
333
334
335 program.Tree.move_ontop(spot, subtree.value().begin());
336
337
338
339 return true;
340 }

417template<Brush::ProgramType T>
418std::optional<Individual<T>> Variation<T>::cross(
419 const Individual<T>& mom, const Individual<T>& dad)
420{
421 /* subtree crossover between this and other, producing new Program */
422 // choose location by weighted sampling of program
423 // TODO: why doesn't this copy the search space reference to child?
424 Program<T> child(mom.program);
425
426 // pick a subtree to replace
427 vector<float> child_weights(child.Tree.size());
428
429 auto child_iter = child.Tree.begin();
430 std::transform(child.Tree.begin(), child.Tree.end(), child_weights.begin(),
431 [&](const auto& n){
432 auto s_at = child.size_at(child_iter);
433 auto d_at = child.depth_to_reach(child_iter);
434
435 std::advance(child_iter, 1);
436
437 if (s_at<parameters.max_size && d_at<parameters.max_depth)
438 return n.get_prob_change();
439 else
440 return 0.0f;
441 }
442 );
443
444 if (std::all_of(child_weights.begin(), child_weights.end(), [](const auto& w) {
445 return w<=0.0;
446 }))
447 { // There is no spot that has a probability to be selected
448 return std::nullopt;
449 }
450
451 // pick a subtree to insert. Selection is based on other_weights
452 Program<T> other(dad.program);
453
454 int attempts = 0;
455 while (++attempts <= 3)
456 {
457 auto child_spot = r.select_randomly(child.Tree.begin(),
458 child.Tree.end(),
459 child_weights.begin(),
460 child_weights.end()
461 );
462
463 auto child_ret_type = child_spot.node->data.ret_type;
464
465 auto allowed_size = parameters.max_size -
466 ( child.size() - child.size_at(child_spot) );
467 auto allowed_depth = parameters.max_depth -
468 ( child.depth_to_reach(child_spot) );
469
470 vector<float> other_weights(other.Tree.size());
471
472 // Iterator to traverse the tree during transformation
473 auto other_iter = other.Tree.begin();
474 std::transform(other.Tree.begin(), other.Tree.end(), other_weights.begin(),
475 [&other, &other_iter, allowed_size, allowed_depth, child_ret_type](const auto& n) mutable {
476 int s = other.size_at(other_iter);
477 int d = other.depth_at(other_iter);
478
479 std::advance(other_iter, 1);
480
481 // Check feasibility and matching return type
482 if (s <= allowed_size && d <= allowed_depth && n.ret_type == child_ret_type) {
483 return n.get_prob_change();
484 }
485
486 return 0.0f; // Non-feasible crossover point
487 }
488 );
489
490 bool matching_spots_found = std::any_of(other_weights.begin(), other_weights.end(),
491 [](float w) { return w > 0.0f; });
492
493 if (matching_spots_found) {
494
495 auto other_spot = r.select_randomly(
496 other.Tree.begin(),
497 other.Tree.end(),
498 other_weights.begin(),
499 other_weights.end()
500 );
501
502 // fmt::print("other_spot : {}\n",other_spot.node->data);
503 // swap subtrees at child_spot and other_spot
504 child.Tree.move_ontop(child_spot, other_spot);
505
506 Individual<T> ind(child);
507 ind.set_variation("cx"); // TODO: use enum here to make it faster
508
509 return ind;
510 }
511 }
512
513 return std::nullopt;

549template<Brush::ProgramType T>
550std::optional<Individual<T>> Variation<T>::mutate(
551 const Individual<T>& parent, string choice)
552{
553 if (choice.empty())
554 {
555
556 auto options = parameters.mutation_probs;
557
558 bool all_zero = true;
559 for (auto &it : parameters.mutation_probs) {
560 if (it.second > 0.0) {
561 all_zero = false;
562 break;
563 }
564 }
565
566 if (all_zero) { // No mutation can be successfully applied to this solution
567 return std::nullopt;
568 }
569
570 // picking a valid mutation option
571 choice = r.random_choice(parameters.mutation_probs);
572 }
573
574 Program<T> copy(parent.program);
575
576 vector<float> weights; // choose location by weighted sampling of program
577 if (choice.compare("point") == 0) // TODO: use enum here to optimize
578 weights = PointMutation::find_spots(copy, (*this), parameters);
579 else if (choice.compare("insert") == 0)
580 weights = InsertMutation::find_spots(copy, (*this), parameters);
581 else if (choice.compare("delete") == 0)
582 weights = DeleteMutation::find_spots(copy, (*this), parameters);
583 else if (choice.compare("subtree") == 0)
584 weights = SubtreeMutation::find_spots(copy, (*this), parameters);
585 else if (choice.compare("toggle_weight_on") == 0)
586 weights = ToggleWeightOnMutation::find_spots(copy, (*this), parameters);
587 else if (choice.compare("toggle_weight_off") == 0)
588 weights = ToggleWeightOffMutation::find_spots(copy, (*this), parameters);
589 else {
590 std::string msg = fmt::format("{} not a valid mutation choice", choice);
591 HANDLE_ERROR_THROW(msg);
592 }
593
594 if (std::all_of(weights.begin(), weights.end(), [](const auto& w) {
595 return w<=0.0;
596 }))
597 { // There is no spot that has a probability to be selected
598 return std::nullopt;
599 }
600 int attempts = 0;
601 while(attempts++ < 3)
602 {
603 Program<T> child(parent.program);
604
605 // apply the mutation and check if it succeeded
606 auto spot = r.select_randomly(child.Tree.begin(), child.Tree.end(),
607 weights.begin(), weights.end());
608
609 // Every mutation here works inplace, so they return bool instead of
610 // std::optional to indicare the result of their manipulation over the
611 // program tree. Here we call the mutation function and return the result
612
613 bool success;
614 if (choice.compare("point") == 0)
615 success = PointMutation::mutate(child, spot, (*this), parameters);
616 else if (choice.compare("insert") == 0)
617 success = InsertMutation::mutate(child, spot, (*this), parameters);
618 else if (choice.compare("delete") == 0)
619 success = DeleteMutation::mutate(child, spot, (*this), parameters);
620 else if (choice.compare("subtree") == 0)
621 success = SubtreeMutation::mutate(child, spot, (*this), parameters);
622 else if (choice.compare("toggle_weight_on") == 0)
623 success = ToggleWeightOnMutation::mutate(child, spot, (*this), parameters);
624 else // it must be"toggle_weight_off"
625 success = ToggleWeightOffMutation::mutate(child, spot, (*this), parameters);
626
627 if (success
628 && ( (child.size() <= parameters.max_size)
629 && (child.depth() <= parameters.max_depth) )){
630
631 Individual<T> ind(child);
632
633 ind.set_variation(choice);
634
635 // subtree performs several samplings, and it will leverate
636 // what point/insert/delete mutations learned about each node utility.
637
638 // TODO: handle subtree - it will sample too many nodes and it may
639 // be hard to track which ones actually improved the expression to
640 // update the bandits/ maybe we should skip it?
641 // mutations that sampled from search space
642 if (choice.compare("point") == 0
643 || choice.compare("insert") == 0
644 || choice.compare("delete") == 0
645 // || choice.compare("subtree") == 0 // TODO: disable this one
646 ) {
647 ind.set_sampled_nodes({spot.node->data});
648 }
649
650 return ind;
651 }
652 else { // reseting
653
654 }
655 }
656
657 return std::nullopt;

660template<Brush::ProgramType T>
661void Variation<T>::vary(Population<T>& pop, int island,
662 const vector<size_t>& parents)
663{
664 auto indices = pop.get_island_indexes(island);
665
666 for (unsigned i = 0; i<indices.size(); ++i)
667 {
668 if (pop.individuals.at(indices.at(i)) != nullptr)
669 {
670 continue; // skipping if it is an individual
671 }
672
673 // pass check for children undergoing variation
674 std::optional<Individual<T>> opt=std::nullopt; // new individual
675
676 const Individual<T>& mom = pop[
677 *r.select_randomly(parents.begin(), parents.end())];
678
679 vector<Individual<T>> ind_parents;
680
681 bool crossover = ( r() < parameters.cx_prob );
682 if (crossover)
683 {
684 const Individual<T>& dad = pop[
685 *r.select_randomly(parents.begin(), parents.end())];
686
687 auto variation_result = cross(mom, dad);
688 ind_parents = {mom, dad};
689 opt = variation_result;
690 }
691 else
692 {
693
694 auto variation_result = mutate(mom);
695
696 ind_parents = {mom};
697 opt = variation_result;
698
699 }
700
701 // this assumes that islands do not share indexes before doing variation
702 unsigned id = parameters.current_gen*parameters.pop_size+indices.at(i);
703
704 // mutation and crossover already perform 3 attempts. If it fails, we just fill with a random individual
705
706 Individual<T> ind;
707 if (opt) // variation worked, lets keep this
708 {
709 ind = opt.value();
710 ind.set_parents(ind_parents);
711 }
712 else { // no optional value was returned. creating a new random individual
713 // It seems that the line below will not fix the root in clf programs
714 ind.init(search_space, parameters); // ind.variation is born by default
715
716 // Program<T> p = search_space.make_program<Program<T>>(parameters, 0, 0);
717 // ind = Individual<T>(p);
718 }
719
720 ind.set_objectives(mom.get_objectives()); // it will have an invalid fitness
721
722 ind.is_fitted_ = false;
723 ind.set_id(id);
724
725 // TODO: smarter way of copying the entire fitness
726 // copying mom fitness to the new individual (without making the fitnes valid)
727 // so the bandits can access this information. Fitness will be valid
728 // only when we do set_values(). We are setting these parameters below
729 // because we want to keep the previous values for the bandits, and
730 // we are not updating the fitness values here.
731 ind.fitness.set_loss(mom.fitness.get_loss());
732 ind.fitness.set_loss_v(mom.fitness.get_loss_v());
733 ind.fitness.set_size(mom.fitness.get_size());
734 ind.fitness.set_complexity(mom.fitness.get_complexity());
735 ind.fitness.set_linear_complexity(mom.fitness.get_linear_complexity());
736 ind.fitness.set_depth(mom.fitness.get_depth());
737
738 // dont set stuff that is not used to calculate the rewards, like crowding_dist
739 // ind.fitness.set_crowding_dist(0.0);
740
741 assert(ind.program.size()>0);
742 assert(ind.fitness.valid()==false);
743
744 pop.individuals.at(indices.at(i)) = std::make_shared<Individual<T>>(ind);
745 }
