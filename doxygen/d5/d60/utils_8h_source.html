153 template<typename T, typename Scalar=T::Scalar>
154 Scalar median(const T& v)
155 {
156  // instantiate a vector
157  vector<Scalar> x(v.size());
158  x.assign(v.data(),v.data()+v.size());
159  // middle element
160  size_t n = x.size()/2;
161  // sort nth element of array
162  nth_element(x.begin(),x.begin()+n,x.end());
163  // if evenly sized, return average of middle two elements
164  if (x.size() % 2 == 0) {
165  nth_element(x.begin(),x.begin()+n-1,x.end());
166  return (x[n] + x[n-1]) / Scalar(2);
167  }
168  // otherwise return middle element
169  else
170  return x[n];
171 };

200 {
201  // initialize original index locations
202  vector<size_t> idx(v.size());
203  std::iota(idx.begin(), idx.end(), 0);
204 
205  // sort indexes based on comparing values in v
206  if (ascending)
207  {
208  sort(idx.begin(), idx.end(),
209  [&v](size_t i1, size_t i2) {return v[i1] < v[i2];});
210  }
211  else
212  {
213  sort(idx.begin(), idx.end(),
214  [&v](size_t i1, size_t i2) {return v[i1] > v[i2];});
215  }
216 
217  return idx;
218 }
