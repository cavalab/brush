631 template<typename P>
632 P SearchSpace::make_program(int max_d, int max_size)
633 {
634  if (max_d == 0)
635  max_d = PARAMS["max_depth"].get<int>();
636  if (max_size == 0)
637  max_size = r.rnd_int(1, PARAMS["max_size"].get<int>());
638 
639  DataType root_type = DataTypeEnum<typename P::TreeType>::value;
640  ProgramType program_type = P::program_type;
641  // ProgramType program_type = ProgramTypeEnum<PT>::value;
642 
643  auto Tree = tree<Node>();
644  if (max_size == 1)
645  {
646  // auto root = Tree.insert(Tree.begin(), sample_terminal(root_type));
647 
648  // We can only have a terminal here, but the terminal must be compatible
649  auto opt = sample_terminal(root_type);
650 
651  if (!opt){
652  auto msg = fmt::format("Program with size=1 could not be created. "
653  "The search space does not contain any terminal with data type {}./n",
654  root_type);
655  HANDLE_ERROR_THROW(msg);
656  }
657 
658  Tree.insert(Tree.begin(), opt.value());
659  }
660  else {// Our program can (and will) be grater than 1 node
661 
662  // building the root node for each program case. We give the root, and it
663  // fills the rest of the tree
664  Node root;
665 
666  // building the root node for each program case
667  if (P::program_type == ProgramType::BinaryClassifier)
668  {
669  root = get(NodeType::Logistic, DataType::ArrayF, Signature<ArrayXf(ArrayXf)>());
670  root.set_prob_change(0.0);
671  root.fixed=true;
672 
673  }
674  else if (P::program_type == ProgramType::MulticlassClassifier)
675  {
676  root = get(NodeType::Softmax, DataType::MatrixF, Signature<ArrayXXf(ArrayXXf)>());
677  root.set_prob_change(0.0);
678  root.fixed=true;
679  }
680  else {
681  // we start with a non-terminal (can be replaced inside PTC2 though, if max_size==1)
682  auto opt = sample_op(root_type);
683  while (!opt) {
684  opt = sample_op(root_type);
685  }
686  root = opt.value();
687  }
688 
689  Tree = PTC2(root, max_d, max_size);
690  }
691 
692  return P(*this,Tree);
