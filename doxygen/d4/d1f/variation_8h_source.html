106 {
107  Program<T> child(parent);
108 
109  // choose location by weighted sampling of program
110  vector<float> weights(child.Tree.size());
111  std::transform(child.Tree.begin(), child.Tree.end(),
112  weights.begin(),
113  [](const auto& n){ return n.get_prob_change(); }
114  );
115 
116  auto spot = r.select_randomly(child.Tree.begin(), child.Tree.end(),
117  weights.begin(), weights.end());
118 
119  auto options = PARAMS["mutation_options"].get<std::map<string,float>>();
120 
121  // Setting to zero the weight of variations that increase the expression
122  // if the expression is already at the maximum size or depth
123  if (child.Tree.size()+1 >= PARAMS["max_size"].get<int>()
124  || child.Tree.max_depth()+1 >= PARAMS["max_depth"].get<int>())
125  {
126  // avoid using mutations that increase size/depth
127  options["insert"] = 0.0;
128  }
129 
130  // choose a valid mutation option
131  string choice = r.random_choice(options);
132 
133  if (choice == "insert")
134  insert_mutation(child.Tree, spot, SS);
135  else if (choice == "delete")
136  delete_mutation(child.Tree, spot, SS);
137  else if (choice == "point")
138  point_mutation(child.Tree, spot, SS);
139  else if (choice == "toggle_weight")
140  toggle_weight_mutation(child.Tree, spot, SS);
141  else{
142  string msg = fmt::format("{} not a valid mutation choice", choice);
143  HANDLE_ERROR_THROW(msg);
144  }
145 
146  return child;
