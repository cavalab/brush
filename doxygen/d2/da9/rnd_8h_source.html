75  Iter select_randomly(Iter start, Iter end, Iter2 wstart, Iter2 wend)
76  {
77  // discrete_distribution creates a Generator for a probability
78  // distribution function. `dis` generate integers from [0, s),
79  // where `s` is the number of probabilities in the iterator
80  // passed as argument. To generate a new value, the
81  // `operator()( Generator& g )` needs a uniform random bit
82  // generator object, which is stored in `rg`.

84  // std::uniform_int_distribution<> dis(0, distance(start, end) - 1);
85  std::discrete_distribution<size_t> dis(wstart, wend);
