505  template<NodeType NT>
506  void MakeNodes(const unordered_map<string,float>& user_ops,
507  const vector<DataType>& unique_data_types
508  )
509  {
510  if (Is<NodeType::Terminal, NodeType::Constant>(NT))
511  return;
512  bool use_all = user_ops.size() == 0;
513  auto name = NodeTypeName.at(NT);
514 
515  // skip operators not defined by user
516  if (!use_all & user_ops.find(name) == user_ops.end())
517  return;
518 
519  using signatures = Signatures<NT>::type;
520  constexpr auto size = std::tuple_size<signatures>::value;
521  AddNodes<NT, signatures>(
522  user_ops,
523  unique_data_types,
524  std::make_index_sequence<size>()
525  );
