59 inline bool insert_mutation(tree<Node>& Tree, Iter spot, const SearchSpace& SS)
60 {
61  // cout << "insert mutation\n";
62  auto spot_type = spot.node->data.ret_type;
63 
64  // pick a random compatible node to insert (with probabilities given by
65  // node_map_weights). The `-1` represents the node being inserted.
66  // Ideally, it should always find at least one match (the same node
67  // used as a reference when calling the function). However, we have a
68  // size restriction, which will be relaxed here (just as it is in the PTC2
69  // algorithm). This mutation can create a new expression that exceeds the
70  // maximum size by the highest arity among the operators.
71  std::optional<Node> n = SS.sample_op_with_arg(spot_type, spot_type, true,
72  PARAMS["max_size"].get<int>()-Tree.size()-1);
73 
74  if (!n) // there is no operator with compatible arguments
75  return false;
76 
77  // make node n wrap the subtree at the chosen spot
78  auto parent_node = Tree.wrap(spot, *n);
79 
80  // now fill the arguments of n appropriately
81  bool spot_filled = false;
82  for (auto a: (*n).arg_types)
83  {
84  if (spot_filled)
85  {
86  // if spot is in its child position, append children.
87  // TODO: reminding that sample_terminal may fail as well
88  auto opt = SS.sample_terminal(a);
89 
90  if (!opt)
91  return false;
92 
93  Tree.append_child(parent_node, opt.value());
94  }
95  // if types match, treat this spot as filled by the spot node
96  else if (a == spot_type)
97  spot_filled = true;
98  // otherwise, add siblings before spot node
99  else {
100  auto opt = SS.sample_terminal(a);
101 
102  if (!opt)
103  return false;
104 
105  Tree.insert(spot, opt.value());
106  }
107  }
108 
109  return true;
110 }

170 {

172  auto max_size = PARAMS["max_size"].get<int>() - (Tree.size() - Tree.size(spot));
173  auto max_depth = PARAMS["max_depth"].get<int>() - (Tree.depth(spot));

222 template<ProgramType T>
223 std::optional<Program<T>> mutate(const Program<T>& parent, const SearchSpace& SS)
224 {
225  // all mutation validation and setup should be done here. Specific mutaiton
226  // functions are intended to work on the program tree thus cannot access
227  // program functions and attributes.
228  Program<T> child(parent);
229 
230  // choose location by weighted sampling of program
231  vector<float> weights(child.Tree.size());
232  std::transform(child.Tree.begin(), child.Tree.end(),
233  weights.begin(),
234  [](const auto& n){ return n.get_prob_change(); }
235  );
236 
237  auto options = PARAMS["mutation_options"].get<std::map<string,float>>();
238 
239  if (std::all_of(weights.begin(), weights.end(), [](const auto& w) {
240  return w<=0.0;
241  }))
242  { // There is no spot that has a probability to be selected
243  return std::nullopt;
244  }
245 
246  auto spot = r.select_randomly(child.Tree.begin(), child.Tree.end(),
247  weights.begin(), weights.end());
248 
249  if (std::all_of(options.begin(), options.end(), [](const auto& kv) {
250  return kv.second<=0.0;
251  }))
252  { // No mutation can be successfully applied to this solution
253  return std::nullopt;
254  }
255 
256  // choose a valid mutation option
257  string choice = r.random_choice(options);
258 
259  // std::cout << "mutation configuration (choice was " << choice << "):" << std::endl;
260  // for (const auto& [k, v] : options)
261  // std::cout << " - " << k << " : " << v << std::endl;
262 
263  // Every mutation here works inplace, so they return bool instead of
264  // std::optional to indicare the result of their manipulation over the
265  // program tree. Here we call the mutation function and return the result
266  using MutationFunc = std::function<bool(tree<Node>&, Iter, const SearchSpace&)>;
267 
268  std::map<std::string, MutationFunc> mutations{
269  {"insert", insert_mutation},
270  {"delete", delete_mutation},
271  {"point", point_mutation},
272  {"subtree", subtree_mutation},
273  {"toggle_weight_on", toggle_weight_on_mutation},
274  {"toggle_weight_off", toggle_weight_off_mutation}
275  };
276 
277  // Try to find the mutation function based on the choice
278  auto it = mutations.find(choice);
279  if (it == mutations.end()) {
280  std::string msg = fmt::format("{} not a valid mutation choice", choice);
281  HANDLE_ERROR_THROW(msg);
282  }
283 
284  // apply the mutation and check if it succeeded
285  bool success = it->second(child.Tree, spot, SS);
286 
287  if (success
288  && ( (child.size() <= PARAMS["max_size"].get<int>() )
289  && (child.depth() <= PARAMS["max_depth"].get<int>()) )){
290 
291  return child;
292  } else {
293  return std::nullopt;
294  }

320 template<ProgramType T>
321 std::optional<Program<T>> cross(const Program<T>& root, const Program<T>& other)
322 {
323  /* subtree crossover between this and other, producing new Program */
324  // choose location by weighted sampling of program
325  // TODO: why doesn't this copy the search space reference to child?
326  Program<T> child(root);
327 
328  // pick a subtree to replace
329  vector<float> child_weights(child.Tree.size());
330  std::transform(child.Tree.begin(), child.Tree.end(),
331  child_weights.begin(),
332  [](const auto& n){ return n.get_prob_change(); }
333  );
334 
335  if (std::all_of(child_weights.begin(), child_weights.end(), [](const auto& w) {
336  return w<=0.0;
337  }))
338  { // There is no spot that has a probability to be selected
339  return std::nullopt;
340  }
341 
342  auto child_spot = r.select_randomly(child.Tree.begin(),
343  child.Tree.end(),
344  child_weights.begin(),
345  child_weights.end()
346  );
347 
348  auto child_ret_type = child_spot.node->data.ret_type;
349 
350  auto allowed_size = PARAMS["max_size"].get<int>() -
351  ( child.size() - child.size_at(child_spot) );
352  auto allowed_depth = PARAMS["max_depth"].get<int>() -
353  ( child.depth_to_reach(child_spot) );
354 
355  // pick a subtree to insert. Selection is based on other_weights
356  vector<float> other_weights(other.Tree.size());
357 
358  // iterator to get the size of subtrees inside transform
359  auto other_iter = other.Tree.begin();
360 
361  // lambda function to check feasibility of solution and increment the iterator
362  const auto check_and_incrm = [other, &other_iter, allowed_size, allowed_depth]() -> bool {
363  int s = other.size_at( other_iter );
364  int d = other.depth_at( other_iter );
365 
366  std::advance(other_iter, 1);
367  return (s <= allowed_size) && (d <= allowed_depth);
368  };
369 
370  std::transform(other.Tree.begin(), other.Tree.end(),
371  other_weights.begin(),
372  [child_ret_type, check_and_incrm](const auto& n){
373  // need to pick a node that has a matching output type to the child_spot.
374  // also need to check if swaping this node wouldn't exceed max_size
375  if (check_and_incrm() && (n.ret_type == child_ret_type))
376  return n.get_prob_change();
377  else
378  // setting the weight to zero to indicate a non-feasible crossover point
379  return float(0.0);
380  }
381  );
382 
383  bool matching_spots_found = false;
384  for (const auto& w: other_weights)
385  {
386  matching_spots_found = w > 0.0;
387 
388  if (matching_spots_found) {
389  auto other_spot = r.select_randomly(
390  other.Tree.begin(),
391  other.Tree.end(),
392  other_weights.begin(),
393  other_weights.end()
394  );
395 
396  // fmt::print("other_spot : {}\n",other_spot.node->data);
397  // swap subtrees at child_spot and other_spot
398  child.Tree.move_ontop(child_spot, other_spot);
399  return child;
400  }
401  }
402 
403  return std::nullopt;
404 };
