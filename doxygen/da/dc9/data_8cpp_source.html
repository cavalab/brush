68 State check_type(const ArrayXf& x)
69 {
70  // get feature types (binary or continuous/categorical)
71  bool isBinary = true;
72  bool isCategorical = true;
73  std::map<float, bool> uniqueMap;
74  for(int i = 0; i < x.size(); i++)
75  {
76 
77  if(x(i) != 0 && x(i) != 1)
78  isBinary = false;
79  if(x(i) != floor(x(i)) && x(i) != ceil(x(i)))
80  isCategorical = false;
81  else
82  uniqueMap[x(i)] = true;
83  }
84 
85  State tmp; // = x;
86 
87  if (isBinary)
88  {
89  tmp = ArrayXb(x.cast<bool>());
90  }
91  else
92  {
93  if(isCategorical && uniqueMap.size() < 10)
94  {
95  tmp = ArrayXi(x.cast<int>());
96  }
97  else
98  {
99  tmp = x;
100  }
101  }
102  return tmp;
103

108 {
109  std::map<std::string, State> new_features;
110  for (auto& [key, value] : this->features)
111  {
112  std::visit([&](auto&& arg)
113  {
114  using T = std::decay_t<decltype(arg)>;
115  if constexpr ( T::NumDimensions == 1)
116  new_features[key] = T(arg(idx));
117  else if constexpr (T::NumDimensions==2)
118  new_features[key] = T(arg(idx, Eigen::all));
119  else
120  static_assert(always_false_v<T>, "non-exhaustive visitor!");
121  },
122  value
123  );
124  }

128  new_y = this->y(idx);
129  }
