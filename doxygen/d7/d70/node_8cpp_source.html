50string Node::get_model(const vector<string>& children) const noexcept

52 if (children.empty())

62 else if (Is<NodeType::SplitOn>(node_type)){

66 return fmt::format("If({},{},{})",

74 children.at(0),
75 W,
76 children.at(1),
77 children.at(2)

86 if (i < children.size()-1)

91 }
